#include <iostream>
#include <list>
using namespace std;

list<unsigned> simpleDigits(int n){
    list<unsigned> result;
    for(int i = 2; i <=n; i++){
        if(i % 2 == 0 && i != 2)
            continue;
        if(i % 3 == 0 && i != 3)
            continue;
        if(i % 5 == 0 && i != 5)
            continue;
        bool flag = false;
        for(auto j = result.begin(); j != result.end(); j++){
            if(i % *j == 0) {
                flag = true;
                break;
            }
        }
        if(flag)
            continue;
        result.push_back(i);
    }
    return result;
}
