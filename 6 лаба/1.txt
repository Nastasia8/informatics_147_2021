#include <iostream>
#include <random>
#define Y 4
#define X 5

using namespace std;

int main(){
    srand(time(0));
    int matrix[Y][X];
    for(int i = 0; i < Y; i++){
        for(int j = 0; j < X; j++){
            matrix[i][j] = rand() % 10;
            cout << matrix[i][j];
        }
        cout << endl;
    }

    //A
    for(int i = 0; i < Y; i++){
        for(int j = 0; j < X/2; j++){
            int buffer = matrix[i][j];
            matrix[i][j] = matrix[i][X - j - 1];
            matrix[i][X - j - 1] = buffer;
        }
    }

    //B
    for(int j = 0; j < X/2;j++){
        for(int i = 0; i < Y; i++){
            int buffer = matrix[i][j];
            matrix[i][j] = matrix[i][X - j - 1];
            matrix[i][X - j - 1] = buffer;
        }
    }

    cout << endl << endl;

    for(int i = 0; i < Y; i++){
        for(int j = 0; j < X; j++){
            cout << matrix[i][j];
        }
        cout << endl;
    }


}
