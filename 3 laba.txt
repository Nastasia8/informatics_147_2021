double sum (int k){
    double sum = 0;
    for(double i = 1; i <= k; i++){
        sum += 2/((2 * i + 1) * (2 * i + 3));
    }
    return sum;
}

double mult(int k){
    double mult = 1;
    for(double i = 1; i <= k; i++){
        mult *= pow(-1, i - 1) + i;
    }

    return mult;
}

    double result = mult(2);

4.#include <iostream>

using namespace std;

int resultFor(int start, int end){
    int sum = 0;
    for(int i = start; i <= end; i++){
        if(i % 2 == 0){
            sum += i;
        }
    }
    return sum;
}

int resultWhile(int start, int end){
    int i = start, sum = 0;

    while(i <= end){
        if(i % 2 == 0){
            sum += i;
        }
        i++;
    }

    return sum;
}


int main() {

    int s = resultWhile(3, 6);
    cout << s;

}

int mult(int value){

        int tmp = i % 10;
        i /= 10;
        arr[j] = tmp;
    }
    if(arr[5] == 0)
        return NULL;
    if(arr[6] != 0){
        return NULL;
    }

}

7.#include <iostream>

void print(int height, int width, char symbol){
    for(int i = 0; i < height; i++){
        for(int j = 0; j < width; j++){
            if(i == 0 || i == height - 1){
                cout << symbol;
            }
            else{
                if (j == 0 || j == width - 1) {
                    cout << symbol;
                } else {
                    cout << ' ';
                }
            }
        }
        cout << endl;
    }
}

int main() {
    int h;
    int w;
    char s;
    cin >> h;
    cin >> w;
    cin >> s;
    print(h,w,s);
}
