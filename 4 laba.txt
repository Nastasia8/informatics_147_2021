void swap(double* a, double* b){

    *a = *b;
    *b = tmp;
}

    double a = 1.5, b = 2.4;
    double* aPtr = &a;

void fillArray(int* array, int size){
    for(int i = 0; i < size; i++){
        array[i] = rand();
    }
}

void show(int* array, int size){
    for(int i = 0; i < size; i++){
        cout << array[i] << ' ';
    }
    cout << endl;
}

int minElem(int* array, int size){
    int min = array[0];
    for(int i = 0; i < size; i++){
        if(min > array[i])
            min = array[i];
    }
    return min;
}

int main() {
    srand(time(0));

    int size;

    cin >> size;

    int* array = new int[size];

    fillArray(array, size);

    show(array, size);

    int res = minElem(array, size);

    cout << res;

}
5.#include <iostream>

using namespace std;


void findMin(int** array, int rows, int cols, int* minElems){
    for(int i = 0; i < rows; i++){
        int min = array[i][0];
        for(int j = 0; j < cols; j++){
            if(min > array[i][j])
                min = array[i][j];
        }
        minElems[i] = min;
    }
}

int main() {

    srand(time(0));


    //CREATING ARRAY
    int rows,cols;
    cin >> rows >> cols;
    int** array = new int*[rows];

    for(int i = 0; i < rows; i++){
        array[i] = new int[cols];
    }


    //ARRAY FILLING
    for(int i = 0; i < rows; i++){
        for(int j = 0; j < cols; j++){
            array[i][j] = rand() % 100 + 1;
        }
    }


    //ARRAY PRINTING
    for(int i = 0; i < rows; i++){
        for(int j = 0; j < cols; j++){
            cout << array[i][j] << ' ';
        }
        cout << endl;
    }

    int* minElems = new int[rows];
    findMin(array, rows, cols, minElems);

    for(int i = 0; i < rows; i++){
        cout << "Min element in row num " << i << " is " << minElems[i] << endl;
    }
    delete[] minElems;

    delete[] array;
}
